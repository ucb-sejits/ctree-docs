    from ctree.c.types import CTypeResolver, NumpyTypeResolver

    for resolver in [CTypeResolver(), NumpyTypeResolver()]:
        ty = resolver.resolve(obj)
        if ty is not None:
            return ty
    raise Exception("Unable to resolve type for %s." % repr(obj))
